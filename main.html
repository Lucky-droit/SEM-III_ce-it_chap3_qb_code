        num = int(input("Enter the number"))
        def fact(num):
            if num == 1 or num==0 :
                return 1 
            else :
                return num*fact(num-1)
        print("factorial of entered number is " + str(fact(num)))

        188 3
        Write a Python function that accepts a string and calculate the number of upper case 
        letters and lower case letters.
        string  = input("Enter String ")
        def counter(string):
            count_of_lowercase = 0 
            count_of_Uppercase  = 0 
            for i in string:
                if 97<=ord(i)<=122:
                    count_of_lowercase += 1 
                elif 65<=ord(i)<=90:
                    count_of_Uppercase +=1 
            if (count_of_Uppercase + count_of_lowercase )!=len(string):
                print("=====>String contains Numbers or Symbols or Spaces ")
            print("Count of lowercase " , count_of_lowercase)
            print("count_of Uppercase " , count_of_Uppercase)
        counter(string)
